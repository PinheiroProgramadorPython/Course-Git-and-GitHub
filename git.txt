Lista de Comandos do Git:

-> git init

-> git add .
git add nome-do-arquivo

-> git branch -M main

-> git status

-> git commit -m "Descrition do Commit"
git commit nome-do-arquivo -m "Descrition do Commit"

-> git remote add origin http://github.com/repositorio
git remote -v

-> git push -u origin main
git push
git push --set-upstream origin "nome-da-branch"

-> git log
git shortlog
git reflog

-> git diff
git diff nome-da-branch
git diff HEAD: nome-do-arquivo
git diff HEAD: /pasta/nome-do-arquivo

-> git pull
git pull origin main

-> git branch
git branch nome-da-branch
git checkout nome-da-branch
git checkout -b "nome-da-branch"
git branch -r
git branch -d nome-da-branch

-> git merge
git merge nome-da-branch
git merge origin/nome-da-branch

-> git restore
git restore nome-do-arquivo
git restore --staged nome-do-arquivo
# A flag --source diz de onde queres puxar o ficheiro. 
# Neste caso, puxa a versão do ficheiro como estava 2 commits atrás (HEAD~2).
git restore --source=HEAD~2 script.js

-> git checkout
git checkout HEAD nome-do-arquivo
git checkout nome-da-branch

-> git switch
git switch nome-da-branch
git switch -c nome-da-branch

-> git stach
git stash List
git stash apply number_id-da-stash
git stash show -p number_id-da-stash

-> git tag
git tag -a nome-da-tag -m "mensagem-sobre-a-versão"
git show nome-da-tag
git checkout nome-da-tag

-> git push --tags
git push origin nome-da-tag
git push origin -d nome-da-tag

-> git clone http://github.com/repositorio
git clone http://github.com/repositorio .

-> git fetch
git fetch origin 
git fetch -a
git fetch -a origin
git fetch --all

-> git gc

-> git reset --hard origin/main
